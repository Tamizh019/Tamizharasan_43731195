## --USING ORDER BY  --
select * from worker where department = 'HR' order by salary desc;
select * from worker order by worker_id desc;

## --USING GROUP BY  --
select department, count(department) from worker group by DEPARTMENT; 
select sum(salary),department from worker group by DEPARTMENT;

## Count employees not in HR or admin ...
select count(first_name) as Employee_count ,department from worker
where department not in ('HR', 'admin') 
group by DEPARTMENT; 

## To find the highest salary in each dept ...
select department,max(salary) as highest_salary from worker
group by department 
order by max(SALARY);

## --USING OFFSET  --

## --1. Find departments with average salary > 100000 --
select department , avg(salary) from worker 
group by department having avg(salary) > 100000 ;

## --2. Find departments where max salary is exactly 500000 --
select department , max(salary) from worker 
group by department having max(SALARY) = 500000;

## --3. Find departments with more than 1 employee and total salary > 100000 --
select department ,sum(salary) from worker 
group by department 
having sum(salary) > 100000 and count(DEPARTMENT)>1;

## --4. Find departments with avg salary between 50000 and 200000 --
select department , avg(salary) from worker 
group by department 
having avg(salary) between 50000 and 200000;
